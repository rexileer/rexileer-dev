from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import List

app = FastAPI()

class DataRequest(BaseModel):
    numbers: List[int]

@app.post("/process")
async def process_numbers(req: DataRequest):
    if not req.numbers:
        raise HTTPException(status_code=400, detail="List of numbers required")
    mean = sum(req.numbers) / len(req.numbers)
    median = sorted(req.numbers)[len(req.numbers)//2]
    unique = len(set(req.numbers))
    return {
        "mean": mean,
        "median": median,
        "unique_count": unique,
        "input": req.numbers
    }
